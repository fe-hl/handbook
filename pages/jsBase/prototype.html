еҺҹеһӢзі»з»ҹж ёеҝғжҰӮеҝө вҖӢ

еҺҹеһӢе®ҡд№ү вҖӢ

  • еҮҪж•°еҜ№иұЎпјҡжүҖжңүеҮҪж•°йғҪжңү prototype еұһжҖ§пјҲжҳҫејҸеҺҹеһӢпјү
  • е®һдҫӢеҜ№иұЎпјҡжүҖжңүеҜ№иұЎйғҪжңү __proto__ еұһжҖ§пјҲйҡҗејҸеҺҹеһӢпјүпјҢжҢҮеҗ‘е…¶жһ„йҖ еҮҪж•°зҡ„ prototype
  • зү№ж®Ҡе…ізі»пјҡ
    • еҮҪж•°зҡ„е®һдҫӢеҜ№иұЎжІЎжңү prototype еұһжҖ§
    • Object.prototype жҳҜжүҖжңүеҜ№иұЎзҡ„еҺҹеһӢз»ҲзӮ№

еҺҹеһӢй“ҫеҪўжҲҗжңәеҲ¶ вҖӢ

еҺҹеһӢй“ҫжҹҘжүҫи§„еҲҷпјҡ

  1. иҮӘиә«жҹҘжүҫпјҡеҜ№иұЎи®ҝй—®еұһжҖ§ж—¶пјҢе…ҲеңЁиҮӘиә«еұһжҖ§дёӯжҹҘжүҫ
  2. еҺҹеһӢжҹҘжүҫпјҡиӢҘжңӘжүҫеҲ°пјҢйҖҡиҝҮ __proto__ еҗ‘дёҠеұӮеҺҹеһӢжҹҘжүҫ
  3. йҖ’еҪ’жҹҘжүҫпјҡйҖ’еҪ’жү§иЎҢжӯӨиҝҮзЁӢпјҢзӣҙеҲ°жүҫеҲ°еұһжҖ§жҲ–еҲ°иҫҫеҺҹеһӢй“ҫйЎ¶з«ҜпјҲnullпјү

constructor еұһжҖ§ вҖӢ

ж ёеҝғзү№жҖ§ вҖӢ

  • constructor жқҘжәҗпјҡе®һдҫӢеҜ№иұЎжң¬иә«жІЎжңү constructor еұһжҖ§
  • жҹҘжүҫжңәеҲ¶пјҡи®ҝй—®ж—¶д»ҺеҺҹеһӢй“ҫжҹҘжүҫпјҢжңҖз»ҲжҢҮеҗ‘жһ„йҖ еҮҪж•°
  • дҪңз”Ёпјҡз”ЁдәҺж ҮиҜҶеҜ№иұЎзҡ„жһ„йҖ еҮҪж•°пјҢдҫҝдәҺзұ»еһӢеҲӨж–ӯ

javascript
function myInstanceof(left, right) {
  // еҹәжң¬зұ»еһӢзӣҙжҺҘиҝ”еӣһfalse
  if (left === null || typeof left !== "object") return false;

  let proto = left.__proto__;
  const prototype = right.prototype;

  while (true) {
    if (proto === null) return false;
    if (proto === prototype) return true;
    proto = proto.__proto__;
  }
}

new ж“ҚдҪңз¬Ұе®һзҺ° вҖӢ

ж ёеҝғжӯҘйӘӨпјҡ

  1. еҲӣе»әеҜ№иұЎпјҡеҲӣе»әж–°еҜ№иұЎе№¶й“ҫжҺҘеҲ°жһ„йҖ еҮҪж•°зҡ„еҺҹеһӢ
  2. жү§иЎҢжһ„йҖ еҮҪж•°пјҡз»‘е®ҡ this е№¶дј йҖ’еҸӮж•°
  3. еӨ„зҗҶиҝ”еӣһеҖјпјҡеҰӮжһңжһ„йҖ еҮҪж•°иҝ”еӣһеҜ№иұЎпјҢдҪҝз”ЁиҜҘеҜ№иұЎпјӣеҗҰеҲҷдҪҝз”Ёж–°еҲӣе»әзҡ„еҜ№иұЎ
javascript
function myNew(Constructor, ...args) {
  // 1. еҲӣе»әж–°еҜ№иұЎе№¶й“ҫжҺҘеҺҹеһӢ
  const obj = Object.create(Constructor.prototype);

  // 2. жү§иЎҢжһ„йҖ еҮҪж•°е№¶з»‘е®ҡthis
  const result = Constructor.apply(obj, args);

  // 3. еӨ„зҗҶиҝ”еӣһеҖј
  return result instanceof Object ? result : obj;
}

// дҪҝз”ЁзӨәдҫӢ
function Person(name) {
  this.name = name;
}
const p = myNew(Person, "Alice");
console.log(p.name); // 'Alice'

Object.create е®һзҺ° вҖӢ

ж ёеҝғеҺҹзҗҶпјҡ

  • дёӯд»Ӣжһ„йҖ еҮҪж•°пјҡеҲӣе»әдёҖдёӘз©әжһ„йҖ еҮҪж•°дҪңдёәдёӯд»Ӣ
  • еҺҹеһӢи®ҫзҪ®пјҡе°Ҷдёӯд»Ӣжһ„йҖ еҮҪж•°зҡ„еҺҹеһӢи®ҫзҪ®дёәдј е…Ҙзҡ„еҺҹеһӢеҜ№иұЎ
  • е®һдҫӢеҲӣе»әпјҡиҝ”еӣһдёӯд»Ӣжһ„йҖ еҮҪж•°зҡ„е®һдҫӢпјҢиҜҘе®һдҫӢзҡ„еҺҹеһӢжҢҮеҗ‘дј е…Ҙзҡ„еҺҹеһӢеҜ№иұЎ
javascript
Object.myCreate = function (proto) {
  function F() {} // дёӯд»Ӣжһ„йҖ еҮҪж•°

4. еӨҡз»§жүҝе®һзҺ° вҖӢ

javascript
function extend(target, ...sources) {
  sources.forEach((source) => {
    // еӨҚеҲ¶е®һдҫӢеұһжҖ§
    Object.getOwnPropertyNames(source.prototype).forEach((prop) => {
      if (prop !== "constructor") {
        target.prototype[prop] = source.prototype[prop];
      }
    });
  });
  return target;
}

// дҪҝз”ЁзӨәдҫӢ
function Flyer() {}
Flyer.prototype.fly = function () {
  console.log("Flying");
};

function Swimmer() {}
Swimmer.prototype.swim = function () {
  console.log("Swimming");
};

function SuperHero() {}
extend(SuperHero, Flyer, Swimmer);

const hero = new SuperHero();
hero.fly(); // 'Flying'
hero.swim(); // 'Swimming'
