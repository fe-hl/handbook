JS 错误捕获 ​

// 全局错误捕获（同步/异步）
window.onerror = (message, sourceURL, line, column, errorObj) => {
  const errorData = {
    type: "js-error",
    message,
    sourceURL,
    line,
    column,
    stack: errorObj?.stack,
  };
  SDK.report(errorData);
};

// 资源加载错误
window.addEventListener(
  "error",
  (event) => {
    if (event.target?.src || event.target?.href) {
      SDK.report({
        type: "resource-error",
        url: event.target.src || event.target.href,
        tagName: event.target.tagName,
      });
    }
  },
  true
);

// Promise异常捕获
window.addEventListener("unhandledrejection", (event) => {
  if (event.reason?.stack) {
    SDK.report({
      type: "promise-error",
      reason: event.reason.toString(),
      stack: event.reason.stack,
    });
  }
});

2. 路由监控系统 ​

路由兼容性处理 ​

typescript
class RouterMonitor {
  private isBeforeunload = false;

  constructor() {
    this.rewriteHistory();
    this.beforeunloadFix();
    this.setupListeners();
  }

  private rewriteHistory() {
    const methods: (keyof History)[] = ["pushState", "replaceState"];
    methods.forEach((method) => {
      const original = history[method];
      history[method] = function (...args) {
        const result = original.apply(this, args);
        const event = new Event(method);
        window.dispatchEvent(event);
        return result;
      };
    });
  }

  private beforeunloadFix() {
    window.addEventListener("beforeunload", () => {
      this.isBeforeunload = true;
    });
  }

  private setupListeners() {
    const events = ["popstate", "pushState", "replaceState", "hashchange"];
    events.forEach((event) => {
      window.addEventListener(event, (e) => this.handleRouteChange(e));
    });
  }

  private handleRouteChange(event: Event) {
    const from = SDK.currentPage;
    const to = window.location.href;

    // 上报路由变化
    SDK.report({
      type: "route-change",
      from,
      to,
      eventType: event.type,
    });

    SDK.currentPage = to;
  }
}

3. 用户行为追踪 ​

曝光埋点实现 ​

typescript
class ExposureTracker {
  private observer: IntersectionObserver;
  private trackedElements = new Set();

  constructor() {
    this.observer = new IntersectionObserver((entries) => {
      entries.forEach((entry) => {
        if (entry.isIntersecting) {
          const target = entry.target;
          const appearKey = target.getAttribute("data-appear");
          if (appearKey && !this.trackedElements.has(target)) {
            SDK.report({
              type: "exposure",
              key: appearKey,
              element: Utils.getSelector(target),
            });
            this.trackedElements.add(target);
          }
        }
      });
    });

    this.init();
  }

  private init() {
    window.addEventListener("load", this.scanDOM.bind(this));
    Utils.onRouteChange(() => this.scanDOM());
  }

  private scanDOM() {
    const elements = document.querySelectorAll("[data-appear]");
    elements.forEach((el) => {
      if (!this.trackedElements.has(el)) {
        this.observer.observe(el);
      }
    });
  }
}

无痕埋点优化 ​

typescript
class AutoTracker {
  constructor() {
    document.addEventListener("click", this.handleClick.bind(this), true);
  }

  private handleClick(event: MouseEvent) {
    let target = event.target as HTMLElement;
    let reportKey = null;

    // 向上查找最近的有 data-report 属性的元素
    while (target && target !== document.documentElement) {
      reportKey = target.getAttribute("data-report");
      if (reportKey) break;
      target = target.parentElement;
    }

    if (reportKey) {
      SDK.report({
        type: "auto-track",
        key: reportKey,
        position: Utils.getPosition(event),
        path: Utils.getEventPath(event),
      });
    }
  }
}

4. 性能监控系统 ​

关键性能指标采集 ​

typescript
class PerformanceMonitor {
  private vitalsData = {
    FCP: 0,
    LCP: 0,
    FID: 0,
    CLS: 0,
    TTFB: 0,
  };

  constructor() {
    this.setupWebVitals();
    this.setupPerformanceTiming();
    window.addEventListener("load", this.reportPerformance.bind(this));
  }

  private setupWebVitals() {
    onFCP((metric) => (this.vitalsData.FCP = metric.value));
    onLCP((metric) => (this.vitalsData.LCP = metric.value));
    onFID((metric) => (this.vitalsData.FID = metric.value));
    onCLS((metric) => (this.vitalsData.CLS = metric.value));
  }

  private setupPerformanceTiming() {
    const timing = performance.timing;
    this.vitalsData.TTFB = timing.responseStart - timing.requestStart;
  }

  private reportPerformance() {
    const timing = performance.timing;

    const perfData = {
      type: "performance",
      ...this.vitalsData,
      DNS: timing.domainLookupEnd - timing.domainLookupStart,
      TCP: timing.connectEnd - timing.connectStart,
      SSL:
        timing.secureConnectionStart > 0
          ? timing.connectEnd - timing.secureConnectionStart
          : 0,
      TTI: timing.domInteractive - timing.fetchStart,
      DCL: timing.domContentLoadedEventEnd - timing.domContentLoadedEventStart,
      L: timing.loadEventEnd - timing.fetchStart,
    };

    SDK.report(perfData);
  }
}

5. 用户链路追踪 ​

typescript
class UserJourney {
  private sessionId = Utils.generateUUID();
  private startTime = Date.now();
  private currentPage = window.location.href;

  constructor() {
    this.trackPageDwellTime();
    this.trackPV();
    this.setupListeners();
  }

  private trackPageDwellTime() {
    Utils.onRouteChange(() => {
      const now = Date.now();
      const duration = now - this.startTime;

      SDK.report({
        type: "page-dwell",
        page: this.currentPage,
        duration,
        sessionId: this.sessionId,
      });

      this.startTime = now;
      this.currentPage = window.location.href;
    });
  }

  private trackPV() {
    setTimeout(() => {
      SDK.report({
        type: "pv",
        page: this.currentPage,
        sessionId: this.sessionId,
      });

      Utils.onRouteChange(() => {
        SDK.report({
          type: "pv",
          page: window.location.href,
          sessionId: this.sessionId,
        });
      });
    }, 100);
  }

  private setupListeners() {
    window.addEventListener("beforeunload", () => {
      const duration = Date.now() - this.startTime;
      SDK.report({
        type: "page-dwell",
        page: this.currentPage,
        duration,
        sessionId: this.sessionId,
      });
    });
  }
}

数据上报优化策略 ​

1. 智能上报机制 ​

typescript
class Reporter {
  private queue: any[] = [];
  private maxRetries = 3;
  private reportUrl = "https://monitor-api.example.com/log";

  constructor() {
    // 定时上报
    setInterval(() => this.flush(), 30000);

    // 页面关闭前上报
    window.addEventListener("beforeunload", () => this.flush());
  }

  report(data: any) {
    this.queue.push({
      ...data,
      timestamp: Date.now(),
      ua: navigator.userAgent,
      url: window.location.href,
    });

    if (this.queue.length >= 10) {
      this.flush();
    }
  }

  private async flush() {
    if (this.queue.length === 0) return;

    const batch = [...this.queue];
    this.queue = [];

    // 尝试使用 sendBeacon
    if (this.trySendBeacon(batch)) return;

    // 回退到 XHR
    await this.sendByXHR(batch);
  }

  private trySendBeacon(data: any[]): boolean {
    if (!navigator.sendBeacon) return false;

    const blob = new Blob([JSON.stringify(data)], { type: "application/json" });

    return navigator.sendBeacon(this.reportUrl, blob);
  }

  private async sendByXHR(data: any[], retry = 0) {
    return new Promise((resolve) => {
      const xhr = new XMLHttpRequest();
      xhr.open("POST", this.reportUrl, true);
      xhr.setRequestHeader("Content-Type", "application/json");

      xhr.onload = () => {
        if (xhr.status >= 200 && xhr.status < 300) {
          resolve(true);
        } else if (retry < this.maxRetries) {
          setTimeout(() => this.sendByXHR(data, retry + 1), 1000 * (retry + 1));
        }
      };

      xhr.onerror = () => {
        if (retry < this.maxRetries) {
          setTimeout(() => this.sendByXHR(data, retry + 1), 1000 * (retry + 1));
        }
      };

      xhr.send(JSON.stringify(data));
    });
  }
}

2. 数据压缩与采样 ​

typescript
class DataProcessor {
  process(data: any) {
    // 1. 敏感信息过滤
    this.filterSensitiveData(data);

    // 2. 采样控制
    if (!this.shouldSample(data)) return null;

    // 3. 数据压缩
    return this.compressData(data);
  }

  private filterSensitiveData(data: any) {
    // 移除密码字段
    if (data.type === "auto-track") {
      if (data.element?.includes("password")) {
        data.value = "[FILTERED]";
      }
    }
  }

  private shouldSample(data: any): boolean {
    // 错误数据 100% 上报
    if (data.type.includes("error")) return true;

    // 性能数据 30% 采样
    if (data.type === "performance") {
      return Math.random() < 0.3;
    }

    // 其他数据 10% 采样
    return Math.random() < 0.1;
  }

  private compressData(data: any) {
    // 简化数据格式
    const compressed = {
      t: data.type, // 类型缩写
      ts: data.timestamp, // 时间戳
      d: {}, // 数据
    };

    // 按类型精简字段
    switch (data.type) {
      case "js-error":
        compressed.d = {
          msg: data.message,
          line: data.line,
          col: data.column,
        };
        break;
      case "performance":
        compressed.d = {
          fcp: data.FCP,
          lcp: data.LCP,
          fid: data.FID,
        };
        break;
      // ...其他类型处理
    }

    return compressed;
  }
}

用户行为录制系统 ​

优化后的行为录制方案 ​

typescript
class BehaviorRecorder {
  private events: any[] = [];
  private lastEventTime = 0;
  private sessionId = Utils.generateUUID();
  private isRecording = false;

  start() {
    if (this.isRecording) return;

    this.isRecording = true;
    this.setupListeners();
    this.recordInitialState();
  }

  stop() {
    this.isRecording = false;
    this.removeListeners();
    this.saveRecording();
  }

  private recordInitialState() {
    this.events.push({
      type: "init",
      html: document.documentElement.outerHTML,
      width: window.innerWidth,
      height: window.innerHeight,
      timestamp: Date.now(),
    });
  }

  private setupListeners() {
    const events = ["click", "input", "scroll", "mousemove", "keydown"];
    events.forEach((event) => {
      document.addEventListener(event, this.recordEvent.bind(this), true);
    });
  }

  private recordEvent(event: Event) {
    const now = Date.now();
    // 节流处理 - 避免高频事件
    if (now - this.lastEventTime < 50) return;

    this.lastEventTime = now;

    const record = {
      type: event.type,
      target: Utils.getSelector(event.target),
      value: this.getValue(event),
      position: Utils.getPosition(event),
      timestamp: now,
    };

    this.events.push(record);
  }

  private getValue(event: Event): any {
    if (event instanceof InputEvent) {
      const target = event.target as HTMLInputElement;
      // 过滤敏感信息
      if (target.type === "password") return "[FILTERED]";
      return target.value;
    }
    return null;
  }

  private saveRecording() {
    if (this.events.length === 0) return;

    const recording = {
      sessionId: this.sessionId,
      url: window.location.href,
      duration: Date.now() - this.events[0].timestamp,
      events: this.events,
    };

    SDK.report({
      type: "behavior-recording",
      data: recording,
    });

    this.events = [];
  }
}

总结 ​

该前端监控 SDK 设计方案提供了一套完整的监控解决方案，具有以下特点：

  1. 全面性：覆盖错误监控、性能监控、用户行为分析等全方面监控需求
  2. 高性能：优化后的数据采集与上报机制，对业务影响极小
  3. 易扩展：模块化设计，支持自定义插件和指标
  4. 强兼容：支持主流框架（Vue/React/Angular）和现代浏览器
  5. 安全可靠：内置隐私保护机制和错误隔离机制

通过该 SDK，开发团队可以实时掌握前端应用运行状态，快速定位和解决问题，提升用户体验和产品稳定性。
